import csv

def convert_csv(input_file, output_file):
    # Open the input CSV file
    with open(input_file, mode='r', newline='') as infile:
        csv_reader = csv.DictReader(infile)
        
        # Define the columns we want to keep
        columns_to_keep = ['link_id', 'src_bus_id', 'dst_bus_id', 'voltage','circuits', 'dc', 'length_m']
        
        # Open the output CSV file
        with open(output_file, mode='w', newline='') as outfile:
            # Create a CSV writer with semicolon delimiter
            csv_writer = csv.DictWriter(outfile, fieldnames=columns_to_keep, delimiter=';')
            
            # Write the header to the output file
            csv_writer.writeheader()
            
            # Write the rows with the selected columns
            for row in csv_reader:
                filtered_row = {col: row[col] for col in columns_to_keep if col in row}
                csv_writer.writerow(filtered_row)

import csv

def fix_csv_header(input_file, output_file):
    with open(input_file, mode='r', newline='', encoding='utf-8') as infile:
        # Read all lines from the input file
        lines = infile.readlines()
        
        # Fix the header by replacing commas with semicolons
        header = lines[0].strip().replace(',', ';')
        
        # Write the new file with semicolons
        with open(output_file, mode='w', newline='', encoding='utf-8') as outfile:
            # Write the fixed header
            outfile.write(header + '\n')
            
            # Process the rest of the rows
            for line in lines[1:]:
                # Replace commas with semicolons in the rows as well
                fixed_line = line.strip().replace(',', ';')
                outfile.write(fixed_line + '\n')
    
    print(f"File with semicolon delimiter saved to {output_file}")

import csv

def process_and_split_rows(input_file, output_file_clean, output_file_missing):
    # Open the input CSV file to read
    with open(input_file, mode='r', newline='', encoding='utf-8') as infile:
        reader = csv.reader(infile, delimiter=';')
        
        # Read all rows into a list
        rows = list(reader)
        
        # Get the header and the rest of the rows
        header = rows[0]
        data_rows = rows[1:]
        
        # Get the indices for the columns we are interested in
        try:
            voltage_idx = header.index('voltage')
            src_location_idx = header.index('src_location')
            dst_location_idx = header.index('dst_location')
            dc_idx = header.index('dc')  # Add index for the 'dc' column
        except ValueError as e:
            print(f"Error: {e}. Please check the column names.")
            return
        
        # Modify the voltage column if needed and separate rows with missing values
        rows_with_missing = []
        rows_without_missing = []
        
        for row in data_rows:
            # If voltage is empty and dc is 'true', set voltage to 800
            if row[voltage_idx] == '' and row[dc_idx].lower() == 't':
                row[voltage_idx] = '800'
            
            # Separate rows with missing values and valid rows
            if (row[voltage_idx] == '' and row[dc_idx].lower() == 'false') or \
               row[src_location_idx] == '' or \
               row[dst_location_idx] == '':
                rows_with_missing.append(row)
            else:
                rows_without_missing.append(row)
        
    # Write the rows without missing values to the clean output file
    with open(output_file_clean, mode='w', newline='', encoding='utf-8') as outfile_clean:
        writer = csv.writer(outfile_clean, delimiter=';')
        writer.writerow(header)  # Write the header
        writer.writerows(rows_without_missing)  # Write valid rows

    # Write the rows with missing values to the missing output file
    with open(output_file_missing, mode='w', newline='', encoding='utf-8') as outfile_missing:
        writer = csv.writer(outfile_missing, delimiter=';')
        writer.writerow(header)  # Write the header
        writer.writerows(rows_with_missing)  # Write rows with missing values

    print(f"Clean data saved to {output_file_clean}")
    print(f"Rows with missing data saved to {output_file_missing}")

def parse_coord(coord_str):
    return tuple(map(int, coord_str.strip("[]").split(';')))

def euclidean_dist(c1, c2):
    return math.sqrt((c1[0] - c2[0]) ** 2 + (c1[1] - c2[1]) ** 2)

def find_closest(coord, ac_coords, forbidden_coords):
    candidates = ac_coords - forbidden_coords

import pandas as pd
import re
import math
from collections import defaultdict, deque

# Parse location string like "[123;456]" to tuple (123, 456)
def parse_location(loc_str):
    return tuple(map(int, re.findall(r'\d+', loc_str)))

# Euclidean distance between two nodes
def euclidean(p1, p2):
    return math.sqrt((p1[0] - p2[0]) ** 2 + (p1[1] - p2[1]) ** 2)

# Build graph from edges
def build_graph(edges):
    graph = defaultdict(set)
    for src, dst in edges:
        graph[src].add(dst)
        graph[dst].add(src)
    return graph

# BFS to find connected component from a node
def get_component(start, graph, visited):
    queue = deque([start])
    component = set()
    while queue:
        node = queue.popleft()
        if node in visited:
            continue
        visited.add(node)
        component.add(node)
        queue.extend(graph[node] - visited)
    return component

#create sparse matrixes to store adjacency, voltage, curcuits and length

import csv
import numpy as np
from scipy.sparse import coo_matrix, save_npz
import os

# Grid and ID params
GRID_WIDTH = 5563
GRID_HEIGHT = 4478
NUM_NODES = GRID_WIDTH * GRID_HEIGHT

# Input CSV path
input_csv = "/Users/kobabendiks/Desktop/UNI/Bachelor Thesis/Datasets/Transmission/entsoe/Links Edited/PowerLines_v6_ClustersS2ProxAdded.csv"

# Lists to store edges and their attributes
rows, cols = [], []
lengths, voltages, circuits = [], [], []

# Convert '[row;col]' string to unique integer node ID
def loc_to_node_id(loc_str):
    try:
        row, col = map(lambda x: int(float(x)), loc_str.strip("[]").split(";"))
        return row * GRID_WIDTH + col
    except Exception as e:
        raise ValueError(f"Invalid location string '{loc_str}': {e}")

# Convert European-format floats (with commas) to proper float
def safe_float(val, default=0.0):
    try:
        return float(str(val).replace(",", "."))
    except:
        return default

# Read and process the CSV
with open(input_csv, mode="r", newline="", encoding="utf-8") as infile:
    reader = csv.DictReader(infile, delimiter=";")

    for line in reader:
        try:
            src_id = loc_to_node_id(line['src_location'])
            dst_id = loc_to_node_id(line['dst_location'])

            length = safe_float(line.get('length_m', 0))
            voltage = safe_float(line.get('voltage', 0))
            circuit = safe_float(line.get('circuits', 1))  # Default 1

            # Undirected: add both (src  dst) and (dst  src)
            for a, b in [(src_id, dst_id), (dst_id, src_id)]:
                rows.append(a)
                cols.append(b)
                lengths.append(length)
                voltages.append(voltage)
                circuits.append(circuit)

        except Exception as e:
            print(f" Skipping row due to error: {e}")

# Create sparse matrices
adjacency = coo_matrix((np.ones(len(rows)), (rows, cols)), shape=(NUM_NODES, NUM_NODES))
length_matrix = coo_matrix((lengths, (rows, cols)), shape=(NUM_NODES, NUM_NODES))
voltage_matrix = coo_matrix((voltages, (rows, cols)), shape=(NUM_NODES, NUM_NODES))
circuit_matrix = coo_matrix((circuits, (rows, cols)), shape=(NUM_NODES, NUM_NODES))

# Save all matrices
base_path = '/Users/kobabendiks/Desktop/UNI/Bachelor Thesis/Datasets/Transmission/entsoe/Links Edited/Matrices/PowerLinesMatrix'
save_npz(f"{base_path}_adjacency.npz", adjacency)
save_npz(f"{base_path}_length.npz", length_matrix)
save_npz(f"{base_path}_voltage.npz", voltage_matrix)
save_npz(f"{base_path}_circuits.npz", circuit_matrix)

print(" All sparse matrices saved successfully.")

 All sparse matrices saved successfully.
