# Step 2: Read the positions from the CSV (semicolon-delimited!)
def get_positions_from_csv(csv_path):
    positions_to_keep = set()
    with open(csv_path, newline='', encoding='utf-8') as csvfile:
        reader = csv.DictReader(csvfile, delimiter=';')
        for row in reader:
            for key in ['src_location', 'dst_location']:
                loc = row.get(key)
                if loc:
                    try:
                        loc = loc.strip().replace('[', '').replace(']', '')
                        row_idx, col_idx = map(int, loc.split(';'))  # split on semicolon
                        positions_to_keep.add((row_idx, col_idx))
                    except Exception as e:
                        print(f"Skipping invalid location {loc}: {e}")
    return positions_to_keep

# Step 3: Filter the array and include additional specified nodes
def filter_array_with_extra_nodes(original_array, positions_to_keep, extra_nodes):
    filtered_array = np.full_like(original_array, None, dtype=object)
    count = 0

    # First, add all nodes from CSV
    for row, col in positions_to_keep:
        if 0 <= row < original_array.shape[0] and 0 <= col < original_array.shape[1]:
            value = original_array[row, col]
            if isinstance(value, list):
                filtered_array[row, col] = value
                count += 1

    # Then, add the extra nodes if they exist in original_array
    for node_label in extra_nodes:
        # Convert label "[row;col]" to tuple (row, col)
        row_str, col_str = node_label.strip('[]').split(';')
        row, col = int(row_str), int(col_str)
        if 0 <= row < original_array.shape[0] and 0 <= col < original_array.shape[1]:
            value = original_array[row, col]
            if isinstance(value, list) and filtered_array[row, col] is None:
                filtered_array[row, col] = value
                count += 1

    print(f" Total list elements kept (including extra nodes): {count}")
    return filtered_array

def count_elements(grid):
    num_integers = 0
    num_none = 0
    num_lists = 0
    distinct_integers = set()  # Set to track distinct integers
    
    # Iterate through the grid to count integers, None, and list elements
    for x in range(grid.shape[0]):
        for y in range(grid.shape[1]):
            element = grid[x, y]
            if isinstance(element, int):  # Check if it's an integer
                num_integers += 1
                distinct_integers.add(element)  # Add integer to the set of distinct integers
            elif element is None:  # Check if it's None
                num_none += 1
            elif isinstance(element, list):  # Check if it's a list
                num_lists += 1
    
    num_distinct_integers = len(distinct_integers)  # The number of unique integers
    
    return num_integers, num_none, num_lists, num_distinct_integers

#Creates substation regions by assigning grid cells to nearest cell with a substation by Euclidean Distance
#also creates a csv containing which substations have been assigned what number

import numpy as np
import pandas as pd
from scipy.ndimage import distance_transform_edt
from tqdm import tqdm

# Load the grid
grid = np.load('/Users/kobabendiks/Desktop/UNI/Bachelor Thesis/Datasets/Finalised/Arrays/Substations_array.npy', allow_pickle=True)

# Step 1: Identify substations and assign region IDs
substation_map = np.full(grid.shape, -1, dtype=int)
region_id = 1
substation_info = []
coords = []

for x in range(grid.shape[0]):
    for y in range(grid.shape[1]):
        if isinstance(grid[x, y], list):
            substation_info.append({'row': x, 'col': y, 'region_id': region_id})
            substation_map[x, y] = region_id
            coords.append((x, y, region_id))
            region_id += 1

# Step 2: Create a labeled grid of substation IDs
# Create a binary mask of where substations are
mask = substation_map > 0

# Compute the distance transform and indices of nearest substations
dist, indices = distance_transform_edt(~mask, return_indices=True)

# Step 3: Map every cell to its nearest substation using indices
rows, cols = substation_map.shape
final_grid = np.zeros_like(substation_map)

for i in range(rows):
    for j in range(cols):
        ni, nj = indices[0, i, j], indices[1, i, j]
        final_grid[i, j] = substation_map[ni, nj]

# Step 4: Save results
np.save('/Users/kobabendiks/Desktop/UNI/Bachelor Thesis/Datasets/Transmission/Regions by Substations/RegionsCarved_Substations_alt2.npy', final_grid)

# Count how many cells belong to each region_id
(unique_ids, counts) = np.unique(final_grid, return_counts=True)
region_sizes = dict(zip(unique_ids, counts))

# Add region size to the substation info
df = pd.DataFrame([
    {
        'location': f'[{entry["row"]};{entry["col"]}]',
        'region_id': entry['region_id'],
        'region_size': region_sizes.get(entry['region_id'], 0)
    }
    for entry in substation_info
])

# Save CSV with region size
df.to_csv(
    '/Users/kobabendiks/Desktop/UNI/Bachelor Thesis/Datasets/Transmission/Regions by Substations/Substation_Region_Map.csv',
    index=False,
    sep=';'
)

print("Fast grid assignment by Euclidean distance completed.")

import numpy as np
import matplotlib.pyplot as plt
from matplotlib.colors import ListedColormap

# === 1. Load the main grid ===
grid = np.load('/Users/kobabendiks/Desktop/UNI/Bachelor Thesis/Datasets/Transmission/Regions by Substations/RegionsCarved_Substations_alt2.npy', allow_pickle=True)

# === 2. Load the reference array ===
reference_array = np.load('/Users/kobabendiks/Desktop/UNI/Bachelor Thesis/Datasets/Finalised/Arrays/Reference_array.npy')

# === 3. Process the grid to get integer IDs ===
id_grid = np.full(grid.shape, -1, dtype=int)

if isinstance(grid[0, 0], list):
    list_to_id = {}
    current_id = 0
    for x in range(grid.shape[0]):
        for y in range(grid.shape[1]):
            if isinstance(grid[x, y], list):
                list_tuple = tuple(grid[x, y])
                if list_tuple not in list_to_id:
                    list_to_id[list_tuple] = current_id
                    current_id += 1
                id_grid[x, y] = list_to_id[list_tuple]
else:
    id_grid = grid.astype(int)

# === 4. Combine with reference array ===
combined_grid = np.where(reference_array == -1, -1, id_grid)

# === 5. Extract last digit, but keep -1 intact ===
def last_digit_or_neg1(x):
    if x == -1:
        return -1
    else:
        return x % 10

last_digit_grid = np.vectorize(last_digit_or_neg1)(combined_grid)

# === 6. Define colors: 09 for digits, black for -1 ===
colors = [
    (1, 0, 0),    # Red for 0
    (0, 1, 0),    # Green for 1
    (0, 0, 1),    # Blue for 2
    (1, 1, 0),    # Yellow for 3
    (1, 0, 1),    # Magenta for 4
    (0, 1, 1),    # Cyan for 5
    (0.5, 0.5, 0.5),  # Gray for 6
    (0.5, 0, 0.5),    # Purple for 7
    (0.5, 0.5, 0),    # Olive for 8
    (0.25, 0.75, 0.25)  # Light Green for 9
]

# Add black for -1 as first color
colors_with_black = [(0, 0, 0)] + colors

# Shift last digit grid by +1 so -1  0 index, 09  110
visual_grid = np.where(last_digit_grid == -1, 0, last_digit_grid + 1)

cmap = ListedColormap(colors_with_black)

# === 7. Display ===
plt.figure(figsize=(10, 8))
plt.imshow(visual_grid, cmap=cmap, interpolation='nearest')
